// ---- TABS ----
function switchTab(name) {
  document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.getElementById('tab-' + name).classList.add('active');
  event.target.classList.add('active');
}

// ---- SEVERITY ----
let currentSev = 'medium';
function selectSev(btn, sev) {
  document.querySelectorAll('.sev-btn').forEach(b => b.classList.remove('selected'));
  btn.classList.add('selected');
  currentSev = sev;
}

// ---- BUG REPORT ----
function generateBugReport() {
  const v = id => document.getElementById(id).value || '—';
  const sevMap = { critical:'🔴 CRÍTICO', high:'🟡 ALTO', medium:'🟣 MÉDIO', low:'🟢 BAIXO' };
  const md = `# 🐛 Bug Report: ${v('bug-title')}

| Campo        | Valor |
|---|---|
| **ID**       | ${v('bug-id')} |
| **Data**     | ${v('bug-date')} |
| **Ambiente** | ${v('bug-env')} |
| **Módulo**   | ${v('bug-module')} |
| **Severidade** | ${sevMap[currentSev]} |

## Pré-condições
${v('bug-pre')}

## Passos para Reproduzir
${v('bug-steps')}

## Resultado Esperado
${v('bug-expected')}

## Resultado Obtido
${v('bug-actual')}

## Evidência
${v('bug-evidence')}

## Notas
${v('bug-notes')}
`;
  showOutput('bug-output', md);
}

function clearBug() {
  ['bug-title','bug-id','bug-date','bug-module','bug-pre','bug-steps','bug-expected','bug-actual','bug-evidence','bug-notes'].forEach(id => {
    document.getElementById(id).value = '';
  });
  document.getElementById('bug-output').classList.remove('visible');
}

let caseStatus = {};
planData.forEach(s => s.cases.forEach(c => { caseStatus[c.id] = 'todo'; }));

function renderPlan() {
  const container = document.getElementById('test-sections');
  container.innerHTML = '';
  planData.forEach((section, si) => {
    const div = document.createElement('div');
    div.className = 'test-section';
    div.innerHTML = `
      <div class="test-section-header" onclick="toggleSection(this)">
        <span class="test-section-num">0${si+1}</span>
        <span class="test-section-title">${section.title}</span>
        <span class="chevron">▼</span>
      </div>
      <div class="test-section-body">
        <ul class="checklist">
          ${section.cases.map(c => `
            <li>
              <div class="check ${caseStatus[c.id] === 'pass' ? 'checked' : ''}" onclick="cycleStatus('${c.id}', this)" title="Clique para marcar"></div>
              <span class="check-text ${caseStatus[c.id] === 'pass' ? 'done' : ''}">[${c.id}] ${c.desc}</span>
              <span class="status-pill ${caseStatus[c.id]}" id="pill-${c.id}">${statusLabel(caseStatus[c.id])}</span>
            </li>
          `).join('')}
        </ul>
      </div>
    `;
    container.appendChild(div);
  });
  updateProgress();
}

function statusLabel(s) {
  return { todo:'TODO', pass:'PASS', fail:'FAIL' }[s] || 'TODO';
}

function cycleStatus(id, checkEl) {
  const states = ['todo','pass','fail'];
  const curr = caseStatus[id];
  const next = states[(states.indexOf(curr)+1) % states.length];
  caseStatus[id] = next;
  checkEl.className = 'check' + (next==='pass' ? ' checked' : '');
  const text = checkEl.nextElementSibling;
  text.className = 'check-text' + (next==='pass' ? ' done' : '');
  const pill = document.getElementById('pill-'+id);
  pill.className = 'status-pill ' + next;
  pill.textContent = statusLabel(next);
  updateProgress();
}

function updateProgress() {
  const total = Object.keys(caseStatus).length;
  const done = Object.values(caseStatus).filter(s => s !== 'todo').length;
  document.getElementById('progress-count').textContent = `${done} / ${total} concluídos`;
  document.getElementById('progress-fill').style.width = `${(done/total)*100}%`;
}

function toggleSection(header) {
  header.classList.toggle('open');
  header.nextElementSibling.classList.toggle('open');
}

function resetPlan() {
  Object.keys(caseStatus).forEach(k => caseStatus[k] = 'todo');
  renderPlan();
  document.getElementById('plan-output').classList.remove('visible');
}

function generatePlanReport() {
  const today = new Date().toLocaleDateString('pt-BR');
  let lines = [`# 📋 Relatório de Testes — SauceDemo Login\n**Data:** ${today}\n`];
  planData.forEach(s => {
    lines.push(`\n## ${s.title}\n`);
    lines.push('| ID | Descrição | Status |');
    lines.push('|---|---|---|');
    s.cases.forEach(c => {
      const st = { todo:'⬜ TODO', pass:'✅ PASS', fail:'❌ FAIL' }[caseStatus[c.id]];
      lines.push(`| ${c.id} | ${c.desc} | ${st} |`);
    });
  });
  const pass = Object.values(caseStatus).filter(s=>s==='pass').length;
  const fail = Object.values(caseStatus).filter(s=>s==='fail').length;
  const todo = Object.values(caseStatus).filter(s=>s==='todo').length;
  lines.push(`\n## Resumo\n| Resultado | Qtd |\n|---|---|\n| ✅ PASS | ${pass} |\n| ❌ FAIL | ${fail} |\n| ⬜ TODO | ${todo} |`);
  showOutput('plan-output', lines.join('\n'));
}

// ---- UTILS ----
function showOutput(id, text) {
  const box = document.getElementById(id);
  box.classList.add('visible');
  box.childNodes[box.childNodes.length-1]?.remove?.();
  const pre = document.createTextNode(text);
  box.appendChild(pre);
}

function copyOutput(id) {
  const box = document.getElementById(id);
  const text = box.innerText.replace('COPIAR\n','');
  navigator.clipboard.writeText(text).then(() => showToast());
}

function showToast() {
  const t = document.getElementById('toast');
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2000);
}
